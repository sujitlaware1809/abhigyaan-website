// Countdown Timer
const targetDate = new Date("2025-09-11T09:30:00");


function updateCountdown() {
  const now = new Date();
  const diff = targetDate - now;

  if (diff <= 0) {
    document.getElementById('countdown').innerHTML = "<span style='color:#ff6666'>Event Started!</span>";
    return;
  }

  const days = Math.floor(diff / (1000 * 60 * 60 * 24));
  const hours = Math.floor((diff / (1000 * 60 * 60)) % 24);
  const minutes = Math.floor((diff / 1000 / 60) % 60);
  const seconds = Math.floor((diff / 1000) % 60);

  document.getElementById('days').textContent = String(days).padStart(2, '0');
  document.getElementById('hours').textContent = String(hours).padStart(2, '0');
  document.getElementById('minutes').textContent = String(minutes).padStart(2, '0');
  document.getElementById('seconds').textContent = String(seconds).padStart(2, '0');
}

setInterval(updateCountdown, 1000);
updateCountdown();

document.addEventListener("DOMContentLoaded", () => {
  const modal = document.getElementById("poster-modal");
  const modalImg = modal.querySelector(".modal-img");
  const caption = document.getElementById("modal-caption");
  const closeModal = modal.querySelector(".modal-close");
  const prevBtn = modal.querySelector(".modal-prev");
  const nextBtn = modal.querySelector(".modal-next");
  const posters = Array.from(document.querySelectorAll(".poster-thumb"));

  let currentIndex = 0;
  let slideshowInterval;

  function showImage(index) {
    const img = posters[index];
    if (!img) return;

    modalImg.src = img.src;
    caption.textContent = img.alt;
    currentIndex = index;
    modal.classList.remove("hidden");
  }

  function nextImage() {
    currentIndex = (currentIndex + 1) % posters.length;
    showImage(currentIndex);
  }

  function prevImage() {
    currentIndex = (currentIndex - 1 + posters.length) % posters.length;
    showImage(currentIndex);
  }

  // Autoplay
  function startSlideshow() {
    slideshowInterval = setInterval(nextImage, 5000); // 5 seconds
  }

  function stopSlideshow() {
    clearInterval(slideshowInterval);
  }

  // Click thumbnails to open modal
  posters.forEach((img, index) => {
    img.addEventListener("click", () => {
      showImage(index);
      stopSlideshow();
      startSlideshow();
    });
  });

  // Close
  closeModal.addEventListener("click", () => {
    modal.classList.add("hidden");
    stopSlideshow();
  });

  modal.addEventListener("click", (e) => {
    if (e.target === modal) {
      modal.classList.add("hidden");
      stopSlideshow();
    }
  });

  // Arrows
  nextBtn.addEventListener("click", () => {
    nextImage();
    stopSlideshow();
  });

  prevBtn.addEventListener("click", () => {
    prevImage();
    stopSlideshow();
  });

  // Keyboard navigation
  document.addEventListener("keydown", (e) => {
    if (modal.classList.contains("hidden")) return;

    if (e.key === "ArrowRight") {
      nextImage();
      stopSlideshow();
    } else if (e.key === "ArrowLeft") {
      prevImage();
      stopSlideshow();
    } else if (e.key === "Escape") {
      modal.classList.add("hidden");
      stopSlideshow();
    }
  });
});
